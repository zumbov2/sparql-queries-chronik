 	PREFIX schema: <http://schema.org/>
        PREFIX xsd: <http://www.w3.org/2001/XMLSchema#>

        SELECT DISTINCT ?date_fmt ?tickertext ?linkType ?linkParam
        WHERE {
        {
            SELECT (SUBSTR(STR(xsd:date(?date)),1,10) AS ?date_fmt) ?tickertext ?linkType ?linkParam (SUBSTR(STR(xsd:date(?date)),1,10) AS ?sortDate)
            WHERE {
            BIND(xsd:date(NOW() - "P90D"^^xsd:dayTimeDuration) AS ?startDate)
            BIND(xsd:date(NOW() + "P1D"^^xsd:dayTimeDuration) AS ?endDateExclusive)

            ############################################################################
            # 1) Geschäftsstände Volksinitiativen
            ############################################################################
            {
                SELECT ?date ?tickertext ?linkType ?linkParam
                WHERE {
                <https://politics.ld.admin.ch/political-rights/popular-initiative/1> <https://cube.link/observationSet> ?refObsSet .
                ?refObsSet <https://cube.link/observation> ?refObs .

                ?refObs <https://politics.ld.admin.ch/political-rights/popular-initiative/id> ?id .
                ?refObs <https://politics.ld.admin.ch/political-rights/popular-initiative/titel> ?titelNode .
                ?refObs <https://politics.ld.admin.ch/political-rights/popular-initiative/region> <https://ld.admin.ch/country/CHE> .

                ?titelNode schema:name ?titel .
                FILTER(LANG(?titel) = "it")

                <https://politics.ld.admin.ch/political-rights/popular-initiative-stat/1> <https://cube.link/observationSet> ?statObsSet .
                ?statObsSet <https://cube.link/observation> ?statObs .

                ?statObs <https://politics.ld.admin.ch/political-rights/popular-initiative-stat/id> ?id .
                ?statObs <https://politics.ld.admin.ch/political-rights/popular-initiative-stat/datum> ?date .
                ?statObs <https://politics.ld.admin.ch/political-rights/popular-initiative-stat/stand> ?stand .

                VALUES ?stand {
                    <https://politics.ld.admin.ch/political-rights/popular-initiative-stat/stand/2>
                    <https://politics.ld.admin.ch/political-rights/popular-initiative-stat/stand/7>
                    <https://politics.ld.admin.ch/political-rights/popular-initiative-stat/stand/10>
                    <https://politics.ld.admin.ch/political-rights/popular-initiative-stat/stand/15>
                    <https://politics.ld.admin.ch/political-rights/popular-initiative-stat/stand/17>
                    <https://politics.ld.admin.ch/political-rights/popular-initiative-stat/stand/30>
                    <https://politics.ld.admin.ch/political-rights/popular-initiative-stat/stand/18>
                    <https://politics.ld.admin.ch/political-rights/popular-initiative-stat/stand/19>
                    <https://politics.ld.admin.ch/political-rights/popular-initiative-stat/stand/20>
                    <https://politics.ld.admin.ch/political-rights/popular-initiative-stat/stand/21>
                    <https://politics.ld.admin.ch/political-rights/popular-initiative-stat/stand/31>
                }

                OPTIONAL {
                    ?stand schema:name ?standLabel .
                    FILTER(LANG(?standLabel) = "it")
                }

                BIND(
                    IF(
                    ?stand IN (
                        <https://politics.ld.admin.ch/political-rights/popular-initiative-stat/stand/2>,
                        <https://politics.ld.admin.ch/political-rights/popular-initiative-stat/stand/10>,
                        <https://politics.ld.admin.ch/political-rights/popular-initiative-stat/stand/15>,
                        <https://politics.ld.admin.ch/political-rights/popular-initiative-stat/stand/17>,
                        <https://politics.ld.admin.ch/political-rights/popular-initiative-stat/stand/30>,
                        <https://politics.ld.admin.ch/political-rights/popular-initiative-stat/stand/18>,
                        <https://politics.ld.admin.ch/political-rights/popular-initiative-stat/stand/19>,
                        <https://politics.ld.admin.ch/political-rights/popular-initiative-stat/stand/20>,
                        <https://politics.ld.admin.ch/political-rights/popular-initiative-stat/stand/21>,
                        <https://politics.ld.admin.ch/political-rights/popular-initiative-stat/stand/31>
                    ),
                    CONCAT(STR(?standLabel), ": ", STR(?titel)),

                    IF(?stand = <https://politics.ld.admin.ch/political-rights/popular-initiative-stat/stand/7>,
                        CONCAT("Iniziativa depositata: ", STR(?titel)),
                        CONCAT("Iniziativa: ", STR(?titel))
                    )
                    )
                    AS ?tickertext
                )

                BIND(STR('popular-initiative') AS ?linkType)
                BIND(STR(?id) AS ?linkParam)
                }
            }

            UNION

            ############################################################################
            # 2) Geschäftsstände Referenden
            ############################################################################
            {
                SELECT ?date ?tickertext ?linkType ?linkParam
                WHERE {
                {
                    SELECT ?date ?stand ?typGroup ?eventKind
                        (COUNT(DISTINCT ?id) AS ?n)
                        (SAMPLE(?id) AS ?sampleId)
                        (SAMPLE(?titel) AS ?sampleTitel)
                        (SAMPLE(?standLabel) AS ?sampleStandLabel)
                    WHERE {
                    {
                        SELECT ?id ?titel ?typGroup ?stand ?standLabel ?eventKind ?date
                        WHERE {
                        {
                            # Normale Ereignisse mit Originaldatum
                            <https://politics.ld.admin.ch/political-rights/referendum/1> <https://cube.link/observationSet> ?refObsSet .
                            ?refObsSet <https://cube.link/observation> ?refObs .

                            ?refObs <https://politics.ld.admin.ch/political-rights/referendum/id> ?id .
                            ?refObs <https://politics.ld.admin.ch/political-rights/referendum/titel> ?titelNode .
                            ?refObs <https://politics.ld.admin.ch/political-rights/referendum/typ> ?typ .

                            # Eidg. Volksinitiativen sind auch Obl. Referenden (weglassen, sonst doppelt)
                            FILTER(
                                ?typ != <https://politics.ld.admin.ch/political-rights/referendum/typ/9> &&
                                ?typ != <https://politics.ld.admin.ch/political-rights/referendum/typ/10>
                            )

                            ?refObs <https://politics.ld.admin.ch/political-rights/referendum/region> <https://ld.admin.ch/country/CHE> .

                            ?titelNode schema:name ?titel .
                            FILTER(LANG(?titel) = "it")

                            VALUES (?typ ?typGroup) {
                                (<https://politics.ld.admin.ch/political-rights/referendum/typ/16> "fak")
                                (<https://politics.ld.admin.ch/political-rights/referendum/typ/17> "fak")
                                (<https://politics.ld.admin.ch/political-rights/referendum/typ/18> "fak")
                                (<https://politics.ld.admin.ch/political-rights/referendum/typ/19> "fak")
                                (<https://politics.ld.admin.ch/political-rights/referendum/typ/20> "fak")
                                (<https://politics.ld.admin.ch/political-rights/referendum/typ/21> "fak")
                                (<https://politics.ld.admin.ch/political-rights/referendum/typ/22> "fak")
                                (<https://politics.ld.admin.ch/political-rights/referendum/typ/23> "fak")
                                (<https://politics.ld.admin.ch/political-rights/referendum/typ/25> "fak")
                                (<https://politics.ld.admin.ch/political-rights/referendum/typ/26> "fak")
                                (<https://politics.ld.admin.ch/political-rights/referendum/typ/27> "fak")

                                (<https://politics.ld.admin.ch/political-rights/referendum/typ/7> "obl")
                                (<https://politics.ld.admin.ch/political-rights/referendum/typ/8> "obl")
                                (<https://politics.ld.admin.ch/political-rights/referendum/typ/9> "obl")
                                (<https://politics.ld.admin.ch/political-rights/referendum/typ/10> "obl")
                                (<https://politics.ld.admin.ch/political-rights/referendum/typ/11> "obl")
                                (<https://politics.ld.admin.ch/political-rights/referendum/typ/13> "obl")
                                (<https://politics.ld.admin.ch/political-rights/referendum/typ/14> "obl")
                                (<https://politics.ld.admin.ch/political-rights/referendum/typ/15> "obl")
                            }

                            <https://politics.ld.admin.ch/political-rights/referendum-stat/1> <https://cube.link/observationSet> ?statObsSet .
                            ?statObsSet <https://cube.link/observation> ?statObs .

                            ?statObs <https://politics.ld.admin.ch/political-rights/referendum-stat/id> ?id .
                            ?statObs <https://politics.ld.admin.ch/political-rights/referendum-stat/datum> ?rawDate .
                            ?statObs <https://politics.ld.admin.ch/political-rights/referendum-stat/stand> ?stand .

                            OPTIONAL {
                                ?stand schema:name ?standLabel .
                                FILTER(LANG(?standLabel) = "it")
                            }

                            VALUES ?stand {
                                <https://politics.ld.admin.ch/political-rights/referendum-stat/stand/5>
                                <https://politics.ld.admin.ch/political-rights/referendum-stat/stand/7>
                                <https://politics.ld.admin.ch/political-rights/referendum-stat/stand/8>
                                <https://politics.ld.admin.ch/political-rights/referendum-stat/stand/18>
                            }

                            FILTER(
                                (?typGroup = "fak" && ?stand IN (
                                    <https://politics.ld.admin.ch/political-rights/referendum-stat/stand/7>,
                                    <https://politics.ld.admin.ch/political-rights/referendum-stat/stand/8>,
                                    <https://politics.ld.admin.ch/political-rights/referendum-stat/stand/18>
                                ))
                                ||
                                (?typGroup = "obl" && ?stand IN (
                                    <https://politics.ld.admin.ch/political-rights/referendum-stat/stand/5>,
                                    <https://politics.ld.admin.ch/political-rights/referendum-stat/stand/18>
                                ))
                            )

                            BIND(xsd:date(?rawDate) AS ?date)
                            BIND(
                                IF(
                                    ?typGroup = "fak" && ?stand = <https://politics.ld.admin.ch/political-rights/referendum-stat/stand/7>,
                                    "ablauf",
                                    "normal"
                                ) AS ?eventKind
                            )
                        }

                        UNION

                        {
                            # Zusätzlicher synthetischer Eintrag: Beginn der Referendumsfrist
                            <https://politics.ld.admin.ch/political-rights/referendum/1> <https://cube.link/observationSet> ?refObsSet .
                            ?refObsSet <https://cube.link/observation> ?refObs .

                            ?refObs <https://politics.ld.admin.ch/political-rights/referendum/id> ?id .
                            ?refObs <https://politics.ld.admin.ch/political-rights/referendum/titel> ?titelNode .
                            ?refObs <https://politics.ld.admin.ch/political-rights/referendum/typ> ?typ .

                            FILTER(
                                ?typ != <https://politics.ld.admin.ch/political-rights/referendum/typ/9> &&
                                ?typ != <https://politics.ld.admin.ch/political-rights/referendum/typ/10>
                            )

                            ?refObs <https://politics.ld.admin.ch/political-rights/referendum/region> <https://ld.admin.ch/country/CHE> .

                            ?titelNode schema:name ?titel .
                            FILTER(LANG(?titel) = "it")

                            VALUES (?typ ?typGroup) {
                                (<https://politics.ld.admin.ch/political-rights/referendum/typ/16> "fak")
                                (<https://politics.ld.admin.ch/political-rights/referendum/typ/17> "fak")
                                (<https://politics.ld.admin.ch/political-rights/referendum/typ/18> "fak")
                                (<https://politics.ld.admin.ch/political-rights/referendum/typ/19> "fak")
                                (<https://politics.ld.admin.ch/political-rights/referendum/typ/20> "fak")
                                (<https://politics.ld.admin.ch/political-rights/referendum/typ/21> "fak")
                                (<https://politics.ld.admin.ch/political-rights/referendum/typ/22> "fak")
                                (<https://politics.ld.admin.ch/political-rights/referendum/typ/23> "fak")
                                (<https://politics.ld.admin.ch/political-rights/referendum/typ/25> "fak")
                                (<https://politics.ld.admin.ch/political-rights/referendum/typ/26> "fak")
                                (<https://politics.ld.admin.ch/political-rights/referendum/typ/27> "fak")
                            }

                            <https://politics.ld.admin.ch/political-rights/referendum-stat/1> <https://cube.link/observationSet> ?statObsSet .
                            ?statObsSet <https://cube.link/observation> ?statObs .

                            ?statObs <https://politics.ld.admin.ch/political-rights/referendum-stat/id> ?id .
                            ?statObs <https://politics.ld.admin.ch/political-rights/referendum-stat/datum> ?rawDate .
                            ?statObs <https://politics.ld.admin.ch/political-rights/referendum-stat/stand> <https://politics.ld.admin.ch/political-rights/referendum-stat/stand/7> .

                            OPTIONAL {
                                <https://politics.ld.admin.ch/political-rights/referendum-stat/stand/7> schema:name ?standLabel .
                                FILTER(LANG(?standLabel) = "it")
                            }

                            BIND(<https://politics.ld.admin.ch/political-rights/referendum-stat/stand/7> AS ?stand)
                            BIND(xsd:date(?rawDate - "P100D"^^xsd:dayTimeDuration) AS ?date)
                            BIND("beginn" AS ?eventKind)
                        }
                        }
                    }
                    }
                    GROUP BY ?date ?stand ?typGroup ?eventKind
                }

                BIND(
                    IF(
                        ?eventKind = "beginn",
                        IF(
                            ?n = 1,
                            CONCAT("Inizio del termine di referendum: ", STR(?sampleTitel)),
                            CONCAT("Inizio del termine di referendum per ", STR(?n), " atti")
                        ),

                        IF(
                            (?typGroup = "obl" && ?stand = <https://politics.ld.admin.ch/political-rights/referendum-stat/stand/5>),
                            IF(
                                ?n = 1,
                                CONCAT("Decreto federale soggetto a referendum obbligatorio: ", STR(?sampleTitel)),
                                CONCAT(STR(?n), " decreti federali soggetti a referendum obbligatorio")
                            ),

                            IF(
                                (?typGroup = "fak" && ?stand = <https://politics.ld.admin.ch/political-rights/referendum-stat/stand/7>),
                                IF(
                                    ?n = 1,
                                    CONCAT("Scadenza del termine di referendum: ", STR(?sampleTitel)),
                                    CONCAT("Scadenza del termine di referendum di ", STR(?n), " atti")
                                ),

                                IF(
                                    ?stand = <https://politics.ld.admin.ch/political-rights/referendum-stat/stand/8>,
                                    CONCAT("Referendum depositato: ", STR(?sampleTitel)),

                                    IF(
                                        ?stand = <https://politics.ld.admin.ch/political-rights/referendum-stat/stand/18>,
                                        CONCAT(STR(?sampleStandLabel), ": ", STR(?sampleTitel)),
                                        STR(?sampleTitel)
                                    )
                                )
                            )
                        )
                    ) AS ?tickertext
                )

                BIND(
                    IF(
                        (?eventKind = "beginn" && ?n > 1)
                        ||
                        ((?typGroup = "obl" && ?stand = <https://politics.ld.admin.ch/political-rights/referendum-stat/stand/5>) && ?n > 1)
                        ||
                        ((?typGroup = "fak" && ?stand = <https://politics.ld.admin.ch/political-rights/referendum-stat/stand/7> && ?eventKind = "ablauf") && ?n > 1),
                        STR('referendum-list'),
                        STR('referendum')
                    ) AS ?linkType
                )

                BIND(
                    IF(
                        (?eventKind = "beginn" && ?n > 1)
                        ||
                        ((?typGroup = "obl" && ?stand = <https://politics.ld.admin.ch/political-rights/referendum-stat/stand/5>) && ?n > 1)
                        ||
                        ((?typGroup = "fak" && ?stand = <https://politics.ld.admin.ch/political-rights/referendum-stat/stand/7> && ?eventKind = "ablauf") && ?n > 1),
                        STR(''),
                        STR(?sampleId)
                    ) AS ?linkParam
                )
                }
            }

            UNION

            ############################################################################
            # 3) Abstimmungen
            ############################################################################
            {
                SELECT ?date ?tickertext ?linkType ?linkParam
                WHERE {
                {
                    SELECT ?date (COUNT(DISTINCT ?id) AS ?n)
                    WHERE {
                    <https://politics.ld.admin.ch/political-rights/popular-vote/1> <https://cube.link/observationSet> ?refObsSet .
                    ?refObsSet <https://cube.link/observation> ?refObs .

                    ?refObs <https://politics.ld.admin.ch/political-rights/popular-vote/id> ?id .
                    ?refObs <https://politics.ld.admin.ch/political-rights/popular-vote/date> ?date .
                    ?refObs <https://politics.ld.admin.ch/political-rights/popular-vote/abstimmungstitel> ?voteTitle .
                    ?refObs <https://politics.ld.admin.ch/political-rights/popular-vote/region> <https://ld.admin.ch/country/CHE> .

                    FILTER(?voteTitle != <https://politics.ld.admin.ch/political-rights/popular-vote/abstimmungstitel/stichfrage>)
                    }
                    GROUP BY ?date
                }

                BIND(
                    CONCAT(
                    "Votazione popolare federale (",
                    STR(?n),
                    " ",
                    IF(?n = 1, "oggetto", "oggetti"),
                    ")"
                    ) AS ?tickertext
                )

                BIND(STR('voting-sunday') AS ?linkType)
                BIND(STR(?date) AS ?linkParam)
                }
            }

            UNION

            ############################################################################
            # 4) Medienmitteilungen Abstimmungen
            ############################################################################
            {
                SELECT ?date ?tickertext ?linkType ?linkParam
                WHERE {
                {
                    SELECT ?id (MAX(?mmDate) AS ?date) (SAMPLE(?voteDate) AS ?votedate)
                    WHERE {
                    <https://politics.ld.admin.ch/political-rights/popular-vote/1> <https://cube.link/observationSet> ?obsSet .
                    ?obsSet <https://cube.link/observation> ?obs .

                    ?obs <https://politics.ld.admin.ch/political-rights/popular-vote/mmId> ?id .
                    ?obs <https://politics.ld.admin.ch/political-rights/popular-vote/mmDate> ?mmDate .
                    ?obs <https://politics.ld.admin.ch/political-rights/popular-vote/date> ?voteDate .
                    ?obs <https://politics.ld.admin.ch/political-rights/popular-vote/region> <https://ld.admin.ch/country/CHE> .

                    FILTER(STRLEN(STR(?id)) > 0 && STRLEN(STR(?mmDate)) > 0)
                    }
                    GROUP BY ?id
                }

BIND(
    CONCAT(
        "Oggetti della votazione popolare d",
        IF(DAY(?votedate)=8 || DAY(?votedate)=11, "ell'", "el "),
        STR(DAY(?votedate)), " ",
        IF(MONTH(?votedate)=1, "gennaio",
        IF(MONTH(?votedate)=2, "febbraio",
        IF(MONTH(?votedate)=3, "marzo",
        IF(MONTH(?votedate)=4, "aprile",
        IF(MONTH(?votedate)=5, "maggio",
        IF(MONTH(?votedate)=6, "giugno",
        IF(MONTH(?votedate)=7, "luglio",
        IF(MONTH(?votedate)=8, "agosto",
        IF(MONTH(?votedate)=9, "settembre",
        IF(MONTH(?votedate)=10, "ottobre",
        IF(MONTH(?votedate)=11, "novembre", "dicembre"))))))))))),
        " ",
        STR(YEAR(?votedate)),
        " fissati"
    ) AS ?tickertext
)

                BIND(STR('news') AS ?linkType)
                BIND(STR(?id) AS ?linkParam)
                }
            }

            UNION

            ############################################################################
            # 5) Medienmitteilungen Volksinitiativen
            ############################################################################
            {
                SELECT ?date ?tickertext ?linkType ?linkParam
                WHERE {
                {
                    SELECT ?id
                        (MAX(?mmDate) AS ?date)
                        (SAMPLE(?zust) AS ?zustand)
                        (COUNT(DISTINCT ?itemId) AS ?n)
                        (SAMPLE(?titelDe) AS ?titel)
                    WHERE {
                    <https://politics.ld.admin.ch/political-rights/popular-initiative/1> <https://cube.link/observationSet> ?obsSet .
                    ?obsSet <https://cube.link/observation> ?obs .

                    ?obs <https://politics.ld.admin.ch/political-rights/popular-initiative/mmId> ?id .
                    ?obs <https://politics.ld.admin.ch/political-rights/popular-initiative/mmDate> ?mmDate .
                    ?obs <https://politics.ld.admin.ch/political-rights/popular-initiative/id> ?itemId .
                    ?obs <https://politics.ld.admin.ch/political-rights/popular-initiative/zustandgekommen> ?zust .
                    ?obs <https://politics.ld.admin.ch/political-rights/popular-initiative/titel> ?titelNode .
                    ?obs <https://politics.ld.admin.ch/political-rights/popular-initiative/region> <https://ld.admin.ch/country/CHE> .

                    ?titelNode schema:name ?titelDe .
                    FILTER(LANG(?titelDe) = "it")

                    FILTER(STRLEN(STR(?id)) > 0 && STRLEN(STR(?mmDate)) > 0)
                    }
                    GROUP BY ?id
                }

                BIND(
                    IF(?n = 1,
                    IF(?zustand = <https://politics.ld.admin.ch/political-rights/popular-initiative/zustandgekommen/ja>,
                        CONCAT("Riuscita dell'iniziativa: ", STR(?titel)),
                        CONCAT("Non riuscita dell'iniziativa: ", STR(?titel))
                    ),
                    IF(?zustand = <https://politics.ld.admin.ch/political-rights/popular-initiative/zustandgekommen/ja>,
                        "Riuscita delle iniziative",
                        "Non riuscita delle iniziative"
                    )
                    ) AS ?tickertext
                )

                BIND(STR('news') AS ?linkType)
                BIND(STR(?id) AS ?linkParam)
                }
            }

            UNION

            ############################################################################
            # 6) Medienmitteilungen Referenden
            ############################################################################
            {
                SELECT ?date ?tickertext ?linkType ?linkParam
                WHERE {
                {
                    SELECT ?id
                        (MAX(?mmDate) AS ?date)
                        (SAMPLE(?zust) AS ?zustand)
                        (COUNT(DISTINCT ?itemId) AS ?n)
                        (SAMPLE(?titelDe) AS ?titel)
                    WHERE {
                    <https://politics.ld.admin.ch/political-rights/referendum/1> <https://cube.link/observationSet> ?obsSet .
                    ?obsSet <https://cube.link/observation> ?obs .

                    ?obs <https://politics.ld.admin.ch/political-rights/referendum/mmId> ?id .
                    ?obs <https://politics.ld.admin.ch/political-rights/referendum/mmDate> ?mmDate .
                    ?obs <https://politics.ld.admin.ch/political-rights/referendum/id> ?itemId .
                    ?obs <https://politics.ld.admin.ch/political-rights/referendum/zustandegekommen> ?zust .
                    ?obs <https://politics.ld.admin.ch/political-rights/referendum/titel> ?titelNode .
                    ?obs <https://politics.ld.admin.ch/political-rights/referendum/region> <https://ld.admin.ch/country/CHE> .

                    ?titelNode schema:name ?titelDe .
                    FILTER(LANG(?titelDe) = "it")

                    FILTER(STRLEN(STR(?id)) > 0 && STRLEN(STR(?mmDate)) > 0)
                    }
                    GROUP BY ?id
                }

                BIND(
                    IF(?n = 1,
                    IF(?zustand = <https://politics.ld.admin.ch/political-rights/referendum/zustandegekommen/ja>,
                        CONCAT("Referendum riuscito: ", STR(?titel)),
                        CONCAT("Non riuscita del referendum: ", STR(?titel))
                    ),
                    IF(?zustand = <https://politics.ld.admin.ch/political-rights/referendum/zustandegekommen/ja>,
                        "Referendum riusciti",
                        "Non riuscita dei referendum"
                    )
                    ) AS ?tickertext
                )

                BIND(STR('news') AS ?linkType)
                BIND(STR(?id) AS ?linkParam)
                }
            }

            FILTER(xsd:date(?date) >= ?startDate)
            FILTER(xsd:date(?date) < ?endDateExclusive)
            }
        }
        }
        ORDER BY DESC(?sortDate) ?tickertext
